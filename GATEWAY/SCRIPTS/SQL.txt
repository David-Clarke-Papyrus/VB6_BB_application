SELECT GPCHTP, GPGRAD, GPPACK, GPLABL, GPSIZE, GPVAR, GPIWKN, GPINYR, GPGRWR, GPBLCK,VRCMGC, GPSRC,GPINYR & GPIWKN as WEEKYR,GPQTY+GPRVQT as QTY,GPAMTP+GPRVAM as VAL,GPLABL & GPSIZE & GPPACK as LABSIZ
FROM (FRGPDL1
INNER JOIN FRPOLL1 ON (FRGPDL1.GPORGZ = FRPOLL1.TPORGZ AND FRGPDL1.GPPOOL = FRPOLL1.TPPOOL)) 
INNER JOIN FRVARL1 ON FRGPDL1.GPORGZ = FRVARL1.VRORGZ  AND FRGPDL1.GPTYPE = FRVARL1.VRTYPE AND FRGPDL1.GPVAR =FRVARL1.VRVAR
